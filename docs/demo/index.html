analogue-time-picker

A simple analog timepicker based on google material UI.

12h mode

                    
timePickerInput({
    inputElement: document.getElementById("m12h"),
    mode: 12,
    time: new Date()
});

24h mode

                    
timePickerInput({
    inputElement: document.getElementById("m24h"),
    mode: 24,
    time: new Date()
});

Render in a modal

                    
document
    .getElementById("modal")
    .addEventListener("click", function () {
        timePickerModal({
            time: new Date()
        });
    });

Render on the page

                    
window.clock = timePicker({
    element: document.getElementById("onPage"),
    time: new Date()
});

Alter size

                    
timePicker({
    element: document.getElementById("tiny"),
    time: new Date(),
    width: "100px"
});
